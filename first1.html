  summarise(mean_math = mean(math, na.rm = TRUE)) %>%
  ggplot(aes(x = year, y = mean_math, color = gender)) +
  geom_line() + geom_point() +
  labs(title = "Math Score Trends by Gender", y = "Mean Math Score", x = "Year")
